function displaySchedule() { let ym = document.getElementById("scheduleMonth").value; // e.g. "2026-05" if(!ym) return; // 1. Bulletproof Filtering: Handle Google Sheets auto-date formatting let filtered = globalData.schedule.filter(s => { let recordYM = String(s.yearMonth); // If it matches exactly or starts with it (handles ISO strings like "2026-05-01T...") if (recordYM.startsWith(ym)) return true; // If Google Sheets converted it to text like "May 2026" try { let d = new Date(recordYM); let formattedD = `${d.getFullYear()}-${String(d.getMonth()+1).padStart(2, '0')}`; return formattedD === ym; } catch(e) { return false; } }); let tbody = document.querySelector("#scheduleTable tbody"); if(!filtered.length) { tbody.innerHTML = "No schedule found for this month."; return; } // 2. Safely parse and render the table tbody.innerHTML = filtered.map(s => { let rowClass = s.person === "UNFILLED" ? "class='unfilled'" : ""; // Extract YYYY-MM-DD Date safely let rawDate = String(s.date); let displayDate = rawDate; if (rawDate.includes('T')) { displayDate = rawDate.split('T')[0]; } else if (!isNaN(new Date(rawDate).getTime())) { let d = new Date(rawDate); displayDate = `${d.getFullYear()}-${String(d.getMonth()+1).padStart(2, '0')}-${String(d.getDate()).padStart(2, '0')}`; } // Extract Time safely (Handling ISO strings and locale formats) const formatTime = (val) => { let str = String(val); if(str.includes('T') || !isNaN(new Date(str).getTime())) { return new Date(str).toLocaleTimeString([], {hour: '2-digit', minute:'2-digit'}); } if(str.includes(', ')) return str.split(', ')[1]; return str; }; return ` ${displayDate} ${s.role} ${s.shift} ${formatTime(s.start)} ${formatTime(s.end)} ${s.person} `; }).join(""); }
